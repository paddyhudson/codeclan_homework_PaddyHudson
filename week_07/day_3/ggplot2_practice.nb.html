ggplot2 Practice

library(tidyverse)
loans <- read_csv("data/loans.csv")

#1.1 #Loan amounts range from ~0 to 35,000 and interest rates from 5 to 25. Higher value loans tend tp be clustered around round numbers, and we have no units for either axis!

loans %>% 
  ggplot() +
  aes(x = loan_amnt, y = int_rate) +
  geom_point() +
  labs(x = "Loan Amount",
       y = "Interest Rate",
       title = "Interest Rate vs Loan Amount")

#1.2 #We can start to see clusters of data points and we can confirm that there is no clear correlation between these two variables.

loans %>% 
  ggplot() +
  aes(x = loan_amnt, y = int_rate) +
  geom_point(alpha = 0.1) +
  labs(x = "Loan Amount",
       y = "Interest Rate",
       title = "Interest Rate vs Loan Amount")

#1.3 #We can see a clear relationship between interest rate and grade.

loans %>% 
  ggplot() +
  aes(x = loan_amnt, y = int_rate, colour = grade) +
  geom_point(alpha = 0.1) +
  labs(x = "Loan Amount",
       y = "Interest Rate",
       title = "Interest Rate vs Loan Amount")

#1.4 #We can see that there are relatively few low grade loans, and that high grade loans tend to be of a lower value.

loans %>% 
  ggplot() +
  aes(x = loan_amnt, y = int_rate, colour = grade) +
  geom_point(alpha = 0.1) +
  labs(x = "Loan Amount",
       y = "Interest Rate",
       title = "Interest Rate vs Loan Amount") +
  facet_wrap(~grade)

#2 #This supports the conclusions drawn in part 1 - higher grades correlate with lower interest rates and loan amounts, although the increase in amount from grade to grade is not consistent.

loans %>% 
  group_by(grade) %>% 
  summarise(mean_loan_amnt = mean(loan_amnt),
            mean_int_rate = mean(int_rate)
            ) %>% 
  ggplot() +
  aes(x = grade,
      y = mean_loan_amnt,
      fill = mean_int_rate
      ) +
  geom_col() +
  labs(x = "Grade",
       y = "Mean Loan Amount",
       title = "Mean Loan Amount per Grade vs Grade",
       fill = "Mean Interest Rate")

#3

loans %>%
  mutate(int_rate_bin = cut_width(int_rate, width = 2)) %>%
  group_by(grade, purpose) %>% 
  mutate(grade_purpose_total = sum(loan_amnt)/1000000000) %>% 
  ungroup() %>% 
  ggplot() +
  aes(y = grade_purpose_total,
      x = int_rate_bin,
      fill = grade,
      alpha = grade_purpose_total) +
  geom_col() +
  facet_wrap(~purpose, scales = "free_y") +
  theme(axis.text.x = element_blank(),
        panel.background = element_rect(fill = "white"),
        panel.border = element_rect(colour = "black", fill = NA)) +
  scale_alpha_continuous(guide = "none") +
  scale_y_continuous(labels = scales::comma) +
  labs(
    y = "Sum of Loan Amounts (billions)",
    x = "Interest Rate",
    title = "Sum of Loan Amounts vs Interest Rate",
    subtitle = "Y-axis scales vary across charts \nTransparency represents sum of loan amount\n",
    fill = "Grade"
  )
